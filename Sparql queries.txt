Menneskeskapte katastrofer i Europa mellom 1900-tallet til 2000-tallet

	WHERE { ?mm_disaster rdf:type owl:Manmade_disaster .
			?mm_disaster owl:place "Europe" .
			?mm_disaster owl:year ?year . 
			?mm_disaster owl:Name ?name .
FILTER(?year < 2000 && ?year > 1900)
}
